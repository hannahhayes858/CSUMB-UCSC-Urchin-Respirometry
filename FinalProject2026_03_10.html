Blanks

When doing respirometry, you typically run a blank chamber that does not have an organism in it to account for any background noise (respiration, photosynthesis) that may be affecting the O2 measurements that cannot be attributed to your study organism. We must calculate the oxygen consumption rates of the blanks and minus them from the actual rates from the other chambers.

#Now we need the blank rate to minus 
#Get the blank rates 
blank1_rate <- Field_Reskp_grouped %>%
  filter(chamber == 4) %>%
  pull(umol.sec)
  

blank2_rate <- Field_Reskp_grouped %>%
  filter(chamber == 8) %>%

Aggregate Function

Grouping are not giving us a mean with in the correct order. Let’s try it using the aggregate function!

#We're going to group it differently this time
aggy <- aggregate(umol.sec ~ Tub*temp_bin*chamber, data=Field_Reskp_tbin, mean)


#minus the blanks 
blank1_rate <- aggy %>%
  slice(-148) %>%
  filter(chamber == 4) %>%
  pull(umol.sec)
  

blank2_rate <- aggy %>%
  filter(chamber == 8) %>%
  pull(umol.sec)

#Create the blank_rate column 
aggy_adj <- aggy %>%
  mutate(
    blank_rate = case_when(
      chamber %in% 1:3 ~ first(blank1_rate),
      chamber %in% 5:7 ~ first(blank2_rate),
      TRUE ~ NA_real_
    )
  )#%>% 
  #filter(!chamber == "4" | !chamber == "8") 

#Minus the blank rate and the umoll to get the adjusted value
aggy_corr <- aggy_adj %>% 
  mutate(O2.sec.corr = umol.sec - blank_rate) 

#Correct for the hour, but we need the mass
aggy_hh <- aggy_corr %>% 
  mutate( 
    masskg = case_when(
      chamber == 1 ~ 20.58/1000,
      chamber == 2 ~ 13.08/1000,
      chamber == 3 ~ 20.54/1000,
      chamber == 4 ~ 1/1000,
      chamber == 5 ~ 13.34/1000, 
      chamber == 6 ~ 18.60/1000, 
      chamber == 7 ~ 17.07/1000, 
      chamber == 8 ~ 1/1000
    )
  ) %>% 
  mutate(umol_cm2_hr = (O2.sec.corr * 3600) / masskg)

#remove NAs
aggy_na <- aggy_hh %>% 
    filter(!is.na(umol_cm2_hr))
#make it positive
  aggy_pos <- aggy_na %>% 
    mutate(umol_cm2_hr = -(umol_cm2_hr)) %>% 
    filter(umol_cm2_hr > 0)
  
#log it for get the rate
   aggy_log <- aggy_pos %>% 
    mutate(rate = log(umol_cm2_hr + 0.01))
